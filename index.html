<script>

let facts=JSON.parse(localStorage.getItem('facts'))||[]


function renderFacts(){

let grid=document.getElementById('factsGrid')

grid.innerHTML=''

facts.forEach(f=>{

let card=document.createElement('div')

card.className='card'

card.innerHTML='<h3>'+f.title+'</h3><p>'+f.fact+'</p>'

grid.appendChild(card)

})

}


function addFact(){

let title=document.getElementById('movieTitle').value

let fact=document.getElementById('movieFact').value

if(!title||!fact)return

facts.push({title,fact})

localStorage.setItem('facts',JSON.stringify(facts))

renderFacts()

}


renderFacts()


function searchMovies(){

let input=document.getElementById('searchInput').value.toLowerCase()

let cards=document.querySelectorAll('#movieRow p')

cards.forEach(c=>{

let parent=c.parentElement

if(c.innerText.toLowerCase().includes(input)){

parent.style.display='block'

}else{

parent.style.display='none'

}

})

}


const cities=['Delhi','Mumbai','Bangalore','Kolkata','Chennai']

const weatherDiv=document.getElementById('weather')


cities.forEach(c=>{

let box=document.createElement('div')

box.className='weather-box'

box.innerHTML='<h3>'+c+'</h3><p>Weather API integration ready</p>'

weatherDiv.appendChild(box)

})


document.getElementById('aqiBox').innerText='AQI API integration ready for major cities'

</script>
